# --- 1. –ü—Ä–∏–µ–º –¥–∞–Ω–Ω—ã—Ö: –ó–∞–≥—Ä—É–∑–∫–∞ —Å—Ç–∞—Ç–µ–π –∏–∑ –ª–æ–∫–∞–ª—å–Ω—ã—Ö JSON-—Ñ–∞–π–ª–æ–≤ ---
# –ü—É—Ç—å –∫ –≤–∞—à–µ–º—É JSON-—Ñ–∞–π–ª—É —Å–æ —Å—Ç–∞—Ç—å—è–º–∏
file_path = './your_articles.json'

# –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –¥–ª—è –∏–∑–≤–ª–µ—á–µ–Ω–∏—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω—ã—Ö –ø–æ–ª–µ–π –∏–∑ –∫–∞–∂–¥–æ–π –∑–∞–ø–∏—Å–∏ —Å—Ç–∞—Ç—å–∏ [23]
def extract_article_metadata(record: dict, metadata: dict) -> dict:
    metadata["article_id"] = record.get("article_id")
    metadata["title"] = record.get("title")
    metadata["author"] = record.get("author")
    metadata["publication_date"] = record.get("publication_date")
    metadata["category"] = record.get("category")
    return metadata

try:
    # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º JSONLoader –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∫–∞–∂–¥–æ–≥–æ –æ–±—ä–µ–∫—Ç–∞ —Å—Ç–∞—Ç—å–∏ [23]
    # jq_schema '.[ ]' –≤—ã–±–∏—Ä–∞–µ—Ç –∫–∞–∂–¥—ã–π –æ–±—ä–µ–∫—Ç –≤ –∫–æ—Ä–Ω–µ–≤–æ–º –º–∞—Å—Å–∏–≤–µ JSON
    # content_key —É–∫–∞–∑—ã–≤–∞–µ—Ç 'full_content' –∫–∞–∫ –æ—Å–Ω–æ–≤–Ω–æ–π —Ç–µ–∫—Å—Ç –¥–ª—è Document [23]
    loader = JSONLoader(
        file_path=file_path,
        jq_schema='.[ ]',
        content_key='full_content',
        metadata_func=extract_article_metadata
    )

    # –ó–∞–≥—Ä—É–∂–∞–µ–º –¥–æ–∫—É–º–µ–Ω—Ç—ã
    articles_documents = loader.load()
    print(f"–®–∞–≥ 1: –ó–∞–≥—Ä—É–∂–µ–Ω–æ {len(articles_documents)} –æ–±—ä–µ–∫—Ç–æ–≤ LangChain Document –∏–∑ JSON-—Ñ–∞–π–ª–∞.")
    if articles_documents:
        print(f"   –ü—Ä–∏–º–µ—Ä —Å–æ–¥–µ—Ä–∂–∏–º–æ–≥–æ –ø–µ—Ä–≤–æ–≥–æ –¥–æ–∫—É–º–µ–Ω—Ç–∞ (–ø–µ—Ä–≤—ã–µ 100 —Å–∏–º–≤–æ–ª–æ–≤): {articles_documents.page_content[:100]}...")
        print(f"   –ü—Ä–∏–º–µ—Ä –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –ø–µ—Ä–≤–æ–≥–æ –¥–æ–∫—É–º–µ–Ω—Ç–∞: {articles_documents.metadata}")
    else:
        print("   –í JSON-—Ñ–∞–π–ª–µ –Ω–µ –Ω–∞–π–¥–µ–Ω–æ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤ –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏. –£–±–µ–¥–∏—Ç–µ—Å—å, —á—Ç–æ 'your_articles.json' –Ω–µ –ø—É—Å—Ç.")
        exit() # –í—ã—Ö–æ–¥–∏–º, –µ—Å–ª–∏ –Ω–µ—Ç –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤ –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏
except FileNotFoundError:
    print(f"–û—à–∏–±–∫–∞: –§–∞–π–ª {file_path} –Ω–µ –Ω–∞–π–¥–µ–Ω. –£–±–µ–¥–∏—Ç–µ—Å—å, —á—Ç–æ 'your_articles.json' –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ —Ç–æ–π –∂–µ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏.")

    print(f"–û—à–∏–±–∫–∞: –ù–µ —É–¥–∞–ª–æ—Å—å –¥–µ–∫–æ–¥–∏—Ä–æ–≤–∞—Ç—å JSON –∏–∑ {file_path}. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ —Ñ–æ—Ä–º–∞—Ç —Ñ–∞–π–ª–∞.")
    exit()

    print(f"–ü—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ JSON: {e}")

# chunk_overlap –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–∏–º–≤–æ–ª–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –±—É–¥—É—Ç –ø–µ—Ä–µ–∫—Ä—ã–≤–∞—Ç—å—Å—è –º–µ–∂–¥—É —Å–æ—Å–µ–¥–Ω–∏–º–∏ —Ñ—Ä–∞–≥–º–µ–Ω—Ç–∞–º–∏ [4, 27, 26]
text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,  # –û–ø—Ç–∏–º–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä —Ñ—Ä–∞–≥–º–µ–Ω—Ç–∞ –∑–∞–≤–∏—Å–∏—Ç –æ—Ç LLM –∏ –¥–∞–Ω–Ω—ã—Ö

    separators=["\n\n", "\n", " ", ""] # –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é, –Ω–æ –º–æ–∂–Ω–æ –Ω–∞—Å—Ç—Ä–æ–∏—Ç—å [3, 4]
